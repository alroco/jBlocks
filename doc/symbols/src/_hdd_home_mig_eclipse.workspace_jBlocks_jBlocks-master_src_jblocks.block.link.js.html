 12 /**
 13  * <p>A Link block displays a clickable text.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Link
 16  * @class Define a Link block to be displayed into a JBlocks.
 17  * @augments Text
 18  */
 19 $.jBlocks("defineBlockClass","Link","Text",{
 20 	/**
 21 	 * Initializes an Input block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Link#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Text#init}:
 27 	 * @param {Function} [options.onClick] a function to be called when the block is clicked.
 28 	 * @param {Object} [options.attributes] an object representing the attributes (keys and values) to be assigned to the clickable text element.
 29 	 * @param {String} [options.iconUrl] the optional icon to be displayed in the link.
 30 	 * @param {String} [options.iconPosition] the <i>left</i> or <i>right</i> position for the icon.
 31 	 * @param {String} [options.noIconOverlay="true"] if set, the link text won't overlay the icon. 
 32 	 * @param {String} [options.textAlign="center"] the text in the button is aligned to the given <i>center</i>, <i>left</i> or <i>right</i> value.
 33 	 * @param {Integer} [options.minTextW=1] if the block virtual width is below this value, the text is not displayed.
 34 	 */
 35 	init: function() {
 36 		this._super.apply(this,arguments);
 37 		this.options=$.extend({
 38 			onClick: function() {},
 39 			attributes: {},
 40 			iconUrl: null,
 41 			iconPosition: "left",
 42 			minTextW: 1,
 43 			noIconOverlay: true,
 44 			backgroundClass: "jblocks-block-green-link",
 45 		},this.options);
 46 	},
 47 	onCreate: function() {
 48 		var $this=this;
 49 		this._super();
 50 		this.content.addClass("jblocks-block-link");
 51 		this.content.addClass(this.options.backgroundClass);
 52 		for(var attr in this.options.attributes)
 53 			this.text.attr(attr,this.options.attributes[attr]);
 54 		this.content.bind($.jBlocks("click"),function() {
 55 			$this.options.onClick.call($this);
 56 		});
 57 		if(this.options.iconUrl) {
 58 			this.iconImage=$("<div/>").css({
 59 				"background-image":"url("+this.options.iconUrl+")",
 60 				"background-position": "center center",
 61 				"background-repeat": "no-repeat",
 62 				"background-size": "contain",				
 63 				position: "absolute",
 64 				padding: 0,
 65 				margin: 0,
 66 				top: 0,
 67 			}).appendTo(this.content);
 68 			if(this.options.iconPosition=="left") {
 69 				this.iconImage.css({
 70 					left: 0,
 71 				});
 72 			} else {
 73 				this.iconImage.css({
 74 					right: 0,
 75 				});
 76 			}
 77 			if(this.options.noIconOverlay) {
 78 				if(this.options.iconPosition=="left") {
 79 					this.options.epadding[3]=1;
 80 				} else {
 81 					this.options.epadding[1]=1;
 82 				}
 83 			}
 84 			this.content.css("text-align",this.options.textAlign);
 85 		}
 86 		this.element.hide();
 87 	},
 88 	display: function() {
 89 		if(this.actualLayout.w<this.options.minTextW)
 90 			this.textVal("");
 91 		else
 92 			this.textVal(this.options.text);
 93 		this.element.show();
 94 		this._super.apply(this,arguments);
 95 		if(this.iconImage) {
 96 			this.iconImage.css({
 97 				width: (this.content.width()/this.actualLayout.w)*this.actualLayout.h,
 98 				height: this.content.height(),
 99 			});
100 		}
101 	},
102 });
