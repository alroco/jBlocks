 12 /**
 13  * <p>A TextBox block wraps a Text block into a box, so it can be maximized or closed.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name TextBox
 16  * @class Define a TextBox block to be displayed into a JBlocks.
 17  * @augments Box
 18  * @see Text
 19  */
 20 $.jBlocks("defineBlockClass","TextBox","Box",{
 21 	/**
 22 	 * Initializes a TextBox block.
 23 	 * The method is invoked by the jBlocks core and must never be called directly.
 24 	 * @name TextBox#init
 25 	 * @function
 26 	 * @param {Object} [layout] see {@link Box#init}
 27 	 * @param {Object} [options] in addition to options defined in {@link Box#init}:
 28 	 * @param {String} [text=""] the text to be displayed in the box.
 29 	 * @param {Boolean} [options.multiline=true] should this text be displayed on a single or several lines.
 30 	 * @param {String} [options.containerTag="p"] the HTML tag enclosing the text.
 31 	 */
 32 	init: function() {
 33 		var $this=this;
 34 		this._super.apply(this,arguments);
 35 		this.options=$.extend({
 36 			text: "",
 37 			multiline: false,
 38 			containerTag: "p",
 39 		},this.options);
 40 	},
 41 	onCreate: function() {
 42 		this._super();
 43         this.textPanel=this.setContent("Text",{
 44         	text: this.options.text,
 45         	containerTag: this.options.containerTag,
 46         	multiline: this.options.multiline,
 47         });
 48 	},
 49 });
