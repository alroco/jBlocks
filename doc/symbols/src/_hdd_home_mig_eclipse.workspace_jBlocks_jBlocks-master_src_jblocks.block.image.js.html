 12 /**
 13  * <p>An Image block displays an image.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Image
 16  * @class Define a Image block to be displayed into a JBlocks.
 17  * @augments Block
 18  */
 19 $.jBlocks("defineBlockClass","Image","Block",{
 20 	/**
 21 	 * Initializes an Image block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Image#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {String} [options.src] the URL of the image to be displayed.
 28 	 * @param {String} [options.stretch="keep-ratio"] describes the way the image is resized according to the block size:<ul>
 29 	 * <li><i>full-extend</i>: the image size is set to match the block one, regardless of the ratio.</li>
 30 	 * <li><i>keep-ratio</i>: the image is set to maximum size fitting inside the block, while keeping its ratio.</li>
 31 	 * <li><i>no-oversampling</i>: like <i>keep-ratio</i> but the image is not stretched over its original size.</li></ul> 
 32 	 * @param {String} [options.onClick] a function to be called when the image is clicked.
 33 	 */
 34 	init: function(block,data) {
 35 		this._super.apply(this,arguments);
 36 		this.options=$.extend({
 37 			onClick: null,
 38 			stretch: "keep-ratio",
 39 		},this.options);
 40 		this.imageLoaded=false;
 41 		this.img=null;
 42 		this.image=new Image();
 43 	},
 44 	onCreate: function() {
 45 		var $this=this;
 46 		this._super.apply(this,arguments);
 47 		if(this.options.onClick) {
 48 			this.content.addClass("jblocks-clickable");
 49 			this.content.bind($.jBlocks("click"),function() {
 50 				$this.options.onClick.call($this);
 51 			});
 52 		}
 53 		this.img=$("<img/>").appendTo(this.content);
 54 		this.image.onload=function() {
 55 			$this.imageLoaded=true;
 56 			$this.img.attr("src",$this.image.src);
 57 			$this.updateImage();
 58 		}
 59 		this.image.src=this.options.src;
 60 	},
 61 	updateImage: function() {
 62 		if(!this.imageLoaded || this.content.width()<=0 || this.content.height()<=0)
 63 			return;
 64 		var width, height, marginTop=0, marginLeft=0;
 65 		if(this.options.stretch=="full-extend") {
 66 			width=this.content.width();
 67 			height=this.content.height();
 68 		} else if(this.options.stretch=="no-oversampling" && this.image.width<this.content.width() && this.image.height<this.content.height()) {
 69 			width=this.image.width;
 70 			height=this.image.height;
 71 			marginTop=(this.content.height()-this.image.height)/2;
 72 			marginLeft=(this.content.width()-this.image.width)/2;
 73 		} else {
 74 			var ratio=this.image.width/this.image.height;
 75 			var contentRatio=this.content.width()/this.content.height();
 76 			if(ratio<contentRatio) {
 77 				width=this.content.height()*ratio;
 78 				height=this.content.height();
 79 				marginTop=0;
 80 				marginLeft=(this.content.width()-width)/2;
 81 			} else {
 82 				width=this.content.width();
 83 				height=this.content.width()/ratio;
 84 				marginLeft=0;
 85 				marginTop=(this.content.height()-height)/2;
 86 			}
 87 		}
 88 		this.img.css({
 89 			width: width,
 90 			height: height,
 91 			"margin-left": marginLeft,
 92 			"margin-top": marginTop,
 93 		});
 94 	},
 95 	display: function(geometry) {
 96 		this.updateImage();
 97 	},
 98 });
