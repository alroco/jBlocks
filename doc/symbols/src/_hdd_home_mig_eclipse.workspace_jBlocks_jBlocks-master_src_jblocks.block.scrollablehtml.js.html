 12 /**
 13  * <p>A ScrollableHtml block displays arbitrary HTML and allow scrolling vertically in it.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name ScrollableHtml
 16  * @class Define a ScrollableHtml block to be displayed into a JBlocks.
 17  * @augments Icon
 18  */
 19 $.jBlocks("defineBlockClass","ScrollableHtml","Block",{
 20 	/**
 21 	 * Initializes a ScrollableHtml block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name ScrollableHtml#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {String} [options.htmlContent=""] the initial HTML code to be inserted.
 28 	 * @param {Boolean} [options.verticalAlign=false] align content vertically if possible.
 29 	 * @param {Integer} [options.scrollAnimation=0] the duration of the scroll animation in milliseconds.
 30 	 */
 31 	init: function() {
 32 		this._super.apply(this,arguments);
 33 		this.options=$.extend({
 34 			htmlContent: "",
 35 			scrollAnimation: 0,
 36 			verticalAlign: false,
 37 		},this.options);
 38 	},
 39 	onCreate: function() {
 40 		var $this=this;
 41 		
 42 		this.scrollView=$("<div/>").css({
 43 		}).appendTo(this.content).html(this.options.htmlContent);
 44 ;
 45 		this.content.css({
 46 			"-webkit-overflow-scrolling": "touch",
 47 			"overflow-y": "auto",
 48 			"overflow-x": "hidden",
 49 		});
 50 		this.content.bind("mousewheel DOMMouseScroll",function(event) {
 51 			if($this.content.height()<$this.scrollView.height())
 52 				event.stopPropagation();
 53 		});
 54 	},
 55 	/**
 56 	 * Update scroller state.
 57 	 * @name ScrollableHtml#checkState
 58 	 * @function
 59 	 */
 60 	checkState: function() {
 61 		this.display();
 62 	},
 63 	display: function(geometry) {
 64 		this.scrollView.css("padding-top",0);
 65 		if(this.options.verticalAlign && this.scrollView.height()<this.content.height())
 66 			this.scrollView.css("padding-top",(this.content.height()-this.scrollView.height())/2);
 67 	},
 68 	/**
 69 	 * Returns the HTML container so content can be modified directly.
 70 	 * @name ScrollableHtml#getHtmlContainer
 71 	 * @function
 72 	 * @type jQuerySet
 73 	 * @returns the HTML container
 74 	 */
 75 	getHtmlContainer:function() {
 76 		return this.scrollView;
 77 	},
 78 	/**
 79 	 * Scroll to the bottom of the HTML content.
 80 	 * @name ScrollableHtml#scrollBottom
 81 	 * @function
 82 	 */
 83 	scrollBottom:function() {
 84 		this.content.scrollTop(this.scrollView.height()-this.content.height());
 85 	},
 86 	/**
 87 	 * Scroll to the top of the HTML content.
 88 	 * @name ScrollableHtml#scrollTop
 89 	 * @function
 90 	 */
 91 	scrollTop:function() {
 92 		this.content.scrollTop(0);
 93 	},
 94 	/**
 95 	 * Scroll to make specified height visible.
 96 	 * @name ScrollableHtml#makeVisible
 97 	 * @function
 98 	 */
 99 	makeVisible:function(top,height) {
100 		if(this.content.scrollTop()>top || this.content.scrollTop()+this.content.height()>top+height) 
101 			this.content.scrollTop(Math.max(0,Math.min(top,this.scrollView.height()-height)));
102 	},
103 });
