 12 /**
 13  * <p>An Icon block displays a small non-expandable 1x1 image .</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Icon
 16  * @class Define an Icon block to be displayed into a JBlocks.
 17  * @augments Block
 18  */
 19 $.jBlocks("defineBlockClass","Icon","Block",{
 20 	/**
 21 	 * Initializes an Icon block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Icon#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {String} [options.onClick] a function to be called when the block is clicked.
 28 	 * @param {String} [options.iconClass] css class used for icon.
 29 	 */
 30 	init: function() {
 31 		this._super.apply(this,arguments);
 32 		$.extend(this.layout,{
 33 			w: 1,
 34 			h: 1,
 35 			expandWidth: 0,
 36 			expandHeight: 0,
 37 		});
 38 		this.options=$.extend({
 39 			iconClass:"jblocks-block-icon",
 40 			onClick: function() {},
 41 		},this.options);
 42 	},
 43 	onCreate: function(){
 44 		var $this=this;
 45 		this.content.addClass(this.options.iconClass);
 46 		this.content.bind($.jBlocks("click"),function(event) {
 47 			$this.options.onClick.call($this);
 48 		});
 49 	},
 50 });
