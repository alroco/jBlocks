 12 /**
 13  * <p>
 14  * A List block holds display a list of items, handling the search and scroll.
 15  * </p>
 16  * <p>
 17  * The constructor should not be invoked directly but {@link JBlocks#newBlock}
 18  * or {@link JBlocks#addNewBlock} should be used instead.
 19  * </p>
 20  * 
 21  * @name List
 22  * @class Define a List block to be displayed into a JBlocks.
 23  * @augments JBlock
 24  */
 25 $.jBlocks("defineBlockClass", "List", "JBlock",
 26 		{
 27 			/**
 28 			 * Initializes a List block. The method is invoked by the jBlocks
 29 			 * core and must never be called directly.
 30 			 * 
 31 			 * @name List#init
 32 			 * @function
 33 			 * @param {Object}
 34 			 *            [layout] see {@link Block#init}.
 35 			 * @param {Object}
 36 			 *            [options] in addition to base class options
 37 			 *            {@link JBlock#init}:
 38 			 * @param {Boolean} [options.haveSearch=false] the list has a Search feature to 
 39 			 *            allow filtering the displayed elements.
 40 			 * @param {Boolean} [options.showSearch=false] if the list has a Search feature, this parameter
 41 			 *            defines whether the search input is initially displayed. 
 42 			 * @param {Function}
 43 			 *            [options.createItem] function(item,li) is called to
 44 			 *            build the HTML content of the given <li/>
 45 			 *            element and must return a unique ID based on <i>item</i>. If <i>null</i> is returned, the item
 46 			 *            is not added to the list.
 47 			 * @param {Array}
 48 			 *            [options.items] an array of items to be displayed
 49 			 *            initially in the list.
 50 			 * @param {Function}
 51 			 *            [options.sortFn=null] function(item1,items2) to sort
 52 			 *            the list.
 53 			 * @param {Function}
 54 			 *            [options.onRefresh=null] function called after a refresh.
 55 			 * @param {Function}
 56 			 *            [options.onItemAdded=null] function(item) called after an item has been added.
 57 			 * @param {Function}
 58 			 *            [options.onItemRemoves=null] function(item) called after an item has been removed.
 59 			 */
 60 			init : function() {
 61 				this._super.apply(this, arguments);
 62 				this.options = $.extend({
 63 					items : [],
 64 					createItem: function(item, li) {
 65 						return null;
 66 					},
 67 					searchItem: null,
 68 					haveSearch: false,
 69 					showSearch: false,
 70 					itemHeightRatio: 0.66, 
 71 				}, this.options);
 72 				this.options.scrollAnimation=400;
 73 				this.items = {};
 74 				this.search = null;
 75 				this.lastSearch=null;
 76 			},
 77 			onCreate : function() {
 78 				var $this=this;
 79 				this._super();
 80 				if(this.options.haveSearch) {
 81 					this.search = this.selfJBlocks.addNewBlock("Input",{
 82 						h: 1,
 83 						w: 1,
 84 						pref: "top left",
 85 						priority: 100,
 86 						expandHeight: 0,
 87 						expandWidth: 2,
 88 						visible: this.options.showSearch,
 89 					},{
 90 						emargin: [0.1,0,0.1,0],
 91 						epadding: [0.1,0,0.1,0],
 92 						blockContentClass: "jblocks-list-search",
 93 					});
 94 					$.jBlocks("log","search onCreate");					
 95 					this.search.element.bind("input", function(event) {
 96 						var searchExpr=$this.search.element.val();
 97 						if(searchExpr!=$this.lastSearch) {
 98 							$this.lastSearch=searchExpr;
 99 							$this.filter();
100 						}
101 						event.stopPropagation();
102 						$this.refresh();
103 					});
104 				}
105 				this.content.addClass("jblocks-block-list");
106 				this.ul = $("<ul/>");
107 				this.list=this.selfJBlocks.addNewBlock("ScrollableHtml",{
108 					h: 1,
109 					w: 1,
110 					pref: "left top",
111 				},{
112 					onCreate: function() {
113 						this.getHtmlContainer().append($this.ul);
114 					},
115 					contentBorderRadius: 0,
116 					margin: [0,0,0,0],
117 					scrollAnimation: 400,
118 					verticalAlign: false,
119 				});
120 				for ( var g in this.options.items) {
121 					var li = $("<li/>");
122 					var key = this.options.createItem.call(this,this.options.items[g], li);
123 					if(key) {
124 						li.appendTo(this.ul);
125 						this.items[key] = {
126 							li : li,
127 							data : this.options.items[g],
128 						};
129 					}
130 				}
131 			},
132 			filter: function(itemKey) {
133 				var items=this.items;
134 				if(arguments.length>0) {
135 					items={};
136 					items[itemKey]=this.items[itemKey];
137 				}
138 				for(var key in items) {
139 					var item=items[key];
140 					if(!this.lastSearch)
141 						item.li.show();
142 					else {
143 						var display=false;
144 						if(this.options.searchItem==null) {
145 							if(item.li.html().replace(/<.*?>/g," ").toLowerCase().indexOf(this.lastSearch.toLowerCase())>=0)
146 								display=true;
147 						} else if(this.options.searchItem(item.data,this.lastSearch))
148 							display=true;
149 						if(display)
150 							item.li.show();
151 						else
152 							item.li.hide();
153 					}
154 				}
155 			},
156 			updateLi: function(li) {
157 				var sizeRef=Math.min(this.jBlocks.geometry.uHeight,this.jBlocks.geometry.uWidth);
158 				var itemHeight=sizeRef*this.options.itemHeightRatio;
159 				var imgHeight=itemHeight*0.8;
160 				var imgMargin=itemHeight*0.1;
161 				var liPadding=0;//sizeRef*0+"px";
162 				li.css({
163 					height: itemHeight,
164 					//"padding-left": liPadding,
165 					//"padding-right": liPadding,
166 					"line-height": itemHeight+"px",
167 				}).find("img").css({
168 					height: imgHeight,
169 					width: "auto",
170 					margin: imgMargin,
171 				});				
172 			},
173 			display : function(geometry) {
174 				this.updateLi(this.ul.children("li"));
175 				this.refresh();
176 			},
177 			/** 
178 			 * Sort the item in the list according to the provided sort function.
179 			 * @name List#sort
180 			 * @function
181 			 */
182 			sort : function() {
183 				if (typeof this.options.sortFn =="function") {
184 					var items = [];
185 					for ( var key in this.items) {
186 						items.push(key);
187 					}
188 					var $this = this;
189 					items.sort(function(k1, k2) {
190 						return $this.options.sortFn($this.items[k1].data,
191 								$this.items[k2].data);
192 					});
193 					for ( var i in items) {
194 						var it = this.items[items[i]];
195 						it.li.detach().appendTo(this.ul);
196 					}
197 				}
198 			},
199 			/** 
200 			 * Add an item to the list.
201 			 * @name List#addItem
202 			 * @function
203 			 * @param {Object} the item data.
204 			 */
205 			addItem : function(myItem) {
206 				var li = $("<li/>");
207 				var key = this.options.createItem(myItem, li);
208 				if(key) {
209 					li.appendTo(this.ul);
210 					this.items[key] = {
211 						li : li,
212 						data : myItem,
213 					};
214 					if(this.anchor)
215 						this.updateLi(li);
216 				}
217 				if(this.options.onItemAdded)
218 					this.options.onItemAdded.call(this,myItem);
219 			},
220 			/** 
221 			 * Remove a list item.
222 			 * @name List#removeItem
223 			 * @function
224 			 * @param {String} the key for the item to be removed.
225 			 */
226 			removeItem : function(itemId) {
227 				if (typeof this.items[itemId] != "undefined") {
228 					var item = this.items[itemId];
229 					item.li.remove();
230 					delete this.items[itemId];
231 					if(this.options.onItemRemoved)
232 						this.options.onItemRemoved.call(this,item);
233 				}
234 			},
235 			/** 
236 			 * Remove all list items.
237 			 * @name List#clearAll
238 			 * @function
239 			 */
240 			clearAll : function() {
241 				this.ul.empty();
242 				this.items = {};
243 			},
244 			refresh : function() {
245 				this.sort();
246 				this.list.checkState();
247 				if(this.options.onRefresh)
248 					this.options.onRefresh.call(this);
249 			},
250 			/** 
251 			 * Display or hide the search field.
252 			 * @name List#toggleSearch
253 			 * @function
254 			 */
255 			toggleSearch: function() {
256 				this.search.layout.visible=!this.search.layout.visible;
257 				this.selfJBlocks.checkLayout();
258 			},
259 			/** 
260 			 * Scroll to the bottom of the list.
261 			 * @name List#scrollBottom
262 			 * @function
263 			 */
264 			scrollBottom:function() {
265 				this.list.scrollBottom();
266 			},
267 			/** 
268 			 * Scroll to the top of the list.
269 			 * @name List#scrollTop
270 			 * @function
271 			 */
272 			scrollTop:function() {
273 				this.list.scrollTop();
274 			},
275 			/** 
276 			 * Shuffles the order of the lines in the list.
277 			 * @name List#shuffle
278 			 * @function
279 			 */
280 			shuffle: function() {
281 				var arr=[];
282 				for(var key in this.items)
283 					arr.push(this.items[key].li.detach());
284 				var i = arr.length;
285 				if (i<=0) return;
286 				while (--i) {
287 					var j;
288 					j=Math.floor(Math.random()*(i+1));
289 					var tmp = arr[i];
290 					arr[i] = arr[j];
291 					arr[j] = tmp;
292 				}
293 				for(var i in arr)
294 					this.ul.append(arr[i]);
295 			},
296 			/** 
297 			 * Ensures the item for the given key is visible in the list.
298 			 * @name List#ensureVisible
299 			 * @function
300 			 */
301 			ensureVisible: function(key) {
302 				var item=this.items[key];
303 				if(!item)
304 					return;
305 				var position=item.li.position();
306 				this.list.makeVisible(position.top,item.li.height());
307 			},
308 		});
