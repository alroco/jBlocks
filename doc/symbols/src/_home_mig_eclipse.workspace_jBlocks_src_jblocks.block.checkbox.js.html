 12 /**
 13  * <p>A Checkbox block displays an boolean input field.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Checkbox
 16  * @class Define a Checkbox block to be displayed into a JBlocks.
 17  * @augments Block
 18  */
 19 $.jBlocks("defineBlockClass","Checkbox","Block",{
 20 	/**
 21 	 * Initializes a Checkbox block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Checkbox#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {Function} [options.onChange] a function to be called after the field value has changed.
 28 	 * @param {Function} [options.text=""] the label to be displayed for the input.
 29 	 * @param {Function} [options.checked=false] the initial input value.
 30 	 */
 31 	init: function() {
 32 		this._super.apply(this,arguments);
 33 		this.options=$.extend({
 34 			checked: false,
 35 			onChange: function() {},
 36 			text: "",
 37 		},this.options);
 38 		this.layout.expandHeight=0;
 39 		this.options.blockContentClass="jblocks-clickable";
 40 	},
 41 	onCreate: function() {
 42 		var $this=this;
 43 		this._super.apply(this);
 44 		var $this=this;
 45 		this.input=$("<div/>").addClass("jblocks-checkbox-ticker").css({
 46 			float: "left",
 47 		}).appendTo(this.content);
 48 		this.text=$("<p/>").text(this.options.text).css({
 49 			float: "left",
 50 			margin: 0,
 51 		}).appendTo(this.content);
 52 		this.content.bind("click",function() {
 53 			$this.options.checked=!$this.options.checked;
 54 			$this.updateState();
 55 			$this.options.onChange.call($this,$this.options.checked);
 56 		});
 57 		this.updateState();
 58 	},
 59 	updateState: function() {
 60 		if(this.options.checked)
 61 			this.input.addClass("jblocks-checkbox-checked");
 62 		else
 63 			this.input.removeClass("jblocks-checkbox-checked");
 64 	},
 65 	display: function() {
 66 		this._super.apply(this,arguments);
 67 		var height=this.content.height();
 68 		this.input.css({
 69 			height: height,
 70 			width: height,
 71 		});
 72 		this.text.css({
 73 			//"font-size": height*0.6+"pt",
 74 			"padding-left": height/4+"px",
 75 			"line-height": height+"px",
 76 		});
 77 	},
 78 	/**
 79 	 * Get or set the input field value.
 80 	 * @param {Boolean} [checked] if omitted, the function returns the current value of the field, otherwise, sets the input value.
 81 	 * @returns the input boolean value.
 82 	 */
 83 	val: function() {
 84 		if(arguments.length>0) {
 85 			var prev=this.options.checked;
 86 			this.options.checked=(!arguments[0])?false:true;
 87 			if(prev!=this.options.checked) {
 88 				this.updateState();
 89 				this.options.onChange.call(this,this.options.checked);				
 90 			}
 91 		}
 92 		return this.options.checked;
 93 	},
 94 });
