 12 /**
 13  * <p>A ScrollableHtml block displays arbitrary HTML and allow scrolling vertically in it.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name ScrollableHtml
 16  * @class Define a ScrollableHtml block to be displayed into a JBlocks.
 17  * @augments Icon
 18  */
 19 $.jBlocks("defineBlockClass","ScrollableHtml","Block",{
 20 	/**
 21 	 * Initializes a ScrollableHtml block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name ScrollableHtml#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {String} [options.htmlContent=""] the initial HTML code to be inserted.
 28 	 * @param {Boolean} [options.verticalAlign=false] align content vertically if possible.
 29 	 * @param {Integer} [options.scrollAnimation=0] the duration of the scroll animation in milliseconds.
 30 	 */
 31 	init: function() {
 32 		this._super.apply(this,arguments);
 33 		this.options=$.extend({
 34 			htmlContent: "",
 35 			scrollAnimation: 0,
 36 			verticalAlign: false,
 37 		},this.options);
 38 	},
 39 	onCreate: function() {
 40 		var $this=this;
 41 		
 42 		var container=this.content;
 43 		// scrolling control 
 44 		this.scrollCtrl=$("<div/>").addClass("jb-scroll-list").appendTo(this.content);
 45 		var scrollbarDiv=$("<div/>").addClass("scrollbar").appendTo(this.scrollCtrl);
 46 		var track=$("<div/>").addClass("track").appendTo(scrollbarDiv);
 47 		var tracktop=$("<div/>").addClass("tracktop").appendTo(track);
 48 		var th=$("<div/>").addClass("thumb").appendTo(track);
 49 		var top=$("<div/>").addClass("top").appendTo(th);
 50 		var bottom=$("<div/>").addClass("bottom").appendTo(th);
 51 		var viewport=$("<div/>").addClass("viewport").appendTo(this.scrollCtrl);
 52 		this.scrollViewport=$("<div/>").addClass("overview").appendTo(viewport);
 53 		container=this.scrollViewport;
 54 		this.content.bind("mousewheel DOMMouseScroll",function(event) {
 55 			var listCtrl=$this.scrollCtrl;
 56 			var desabled=scrollbarDiv.hasClass("disable");
 57 			if(desabled)
 58 				return;
 59 			var delta = event.detail < 0 || event.wheelDelta > 0 ? 1 : -1;
 60 			var thumb = th;
 61 			var thumbPos=thumb.position();
 62 			if(delta>0 && thumbPos.top==0)
 63 				return;
 64 			if(delta<0 && thumbPos.top+thumb.height()>=track.height())
 65 				return;
 66 			event.stopPropagation();
 67 		});
 68 		container.html(this.options.htmlContent);
 69 		
 70 		this.scrollCtrl.tinyscrollbar({
 71 			animation: this.options.scrollAnimation,
 72 		}).bind("updatelayout",function(e){$this.checkState();});
 73 	},
 74 	/**
 75 	 * Update scroller state.
 76 	 * @name ScrollableHtml#checkState
 77 	 * @function
 78 	 */
 79 	checkState: function() {
 80 		var height=this.content.height()-(this.scrollViewport.outerHeight(true)-this.scrollViewport.height());
 81 		this.scrollViewport.css("min-height",0);
 82 		this.viewportHeight=this.scrollViewport.height();
 83 		this.scrollViewport.css("min-height",height);
 84 		if (this.scrollCtrl.position()){
 85 			var h=this.scrollCtrl.parent().outerHeight()-this.scrollCtrl.position().top;
 86 			var viewport=this.scrollCtrl.children(".viewport");
 87 
 88 			//var desabledBefore=this.scrollCtrl.children(".scrollbar").hasClass("disable");
 89 			
 90 			viewport.css({height:h});
 91 			try {
 92 				this.scrollCtrl.css({height:h});
 93 				this.scrollCtrl.tinyscrollbar_update();
 94 			} catch(e) {
 95 				this.log("error updating tinyscrollbar "+this.jBlocks.jBlocksIndex);
 96 			}
 97 						
 98 			var geomwidth=this.scrollCtrl.parent().outerWidth();
 99 			var desabled=this.scrollCtrl.children(".scrollbar").hasClass("disable");
100 			var lvWidth=desabled?geomwidth:geomwidth-15;
101 			this.scrollCtrl.css({width:geomwidth});
102 			viewport.css({width:lvWidth});
103 			this.scrollViewport.css({width:lvWidth});
104 			
105 			this.scrollCtrl.tinyscrollbar_update();
106 		}
107 	},
108 	display: function(geometry) {
109 		this.scrollViewport.css("padding-top",0);
110 		this.checkState();
111 		var spareHeight=this.scrollCtrl.height()-this.viewportHeight;
112 		if(this.options.verticalAlign && spareHeight>0)
113 			this.scrollViewport.css("padding-top",spareHeight/2);
114 	},
115 	/**
116 	 * Returns the HTML container so content can be modified directly.
117 	 * @name ScrollableHtml#getHtmlContainer
118 	 * @function
119 	 * @type jQuerySet
120 	 * @returns the HTML container
121 	 */
122 	getHtmlContainer:function() {
123 		return this.scrollViewport;
124 	},
125 	/**
126 	 * Scroll to the bottom of the HTML content.
127 	 * @name ScrollableHtml#scrollBottom
128 	 * @function
129 	 */
130 	scrollBottom:function() {
131 		this.scrollCtrl.tinyscrollbar_update('bottom');
132 	},
133 	/**
134 	 * Scroll to the top of the HTML content.
135 	 * @name ScrollableHtml#scrollTop
136 	 * @function
137 	 */
138 	scrollTop:function() {
139 		this.scrollCtrl.tinyscrollbar_update(0);
140 	},
141 	/**
142 	 * Scroll to make specified height visible.
143 	 * @name ScrollableHtml#makeVisible
144 	 * @function
145 	 */
146 	makeVisible:function(top,height) {
147 		var viewport=this.scrollCtrl.children(".viewport");
148 		var overview=viewport.children(".overview");
149 		var overviewTop=-overview.position().top;
150 		if(top<overviewTop || top+height>overviewTop+viewport.height()) {
151 			var top0=top-(viewport.height()-height)/2;
152 			top0=Math.max(0,top0);
153 			top0=Math.min(overview.height()-viewport.height(),top0);
154 			this.scrollCtrl.tinyscrollbar_update(top0);
155 		}
156 	},
157 });
