 12 /**
 13  * <p>An HtmlElement block displays a single HTML element, controlling size, padding and margin.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name HtmlElement
 16  * @class Define an HtmlElementblock to be displayed into a JBlocks.
 17  * @augments Block
 18  */
 19 $.jBlocks("defineBlockClass","HtmlElement","Block",{
 20 	/**
 21 	 * @name HtmlElement#element
 22 	 * @field
 23 	 */
 24 	element: null,
 25 	/**
 26 	 * Initializes an HtmlElement block.
 27 	 * The method is invoked by the jBlocks core and must never be called directly.
 28 	 * @name HtmlElement#init
 29 	 * @function
 30 	 * @param {Object} [layout] see {@link Block#init}.
 31 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 32 	 * @param {String} [tag="p"] the tag for the HTML element.
 33 	 * @param {Array} [options.emargin=[0,0,0,0]] To be documented.
 34 	 * @param {Array} [options.epadding=[0,0,0,0]] To be documented.
 35 	 */
 36 	init: function() {
 37 		this._super.apply(this,arguments);
 38 		this.options=$.extend({
 39 			emargin: [0,0,0,0],
 40 			epadding: [0,0,0,0],
 41 			tag: "p",
 42 		},this.options);
 43 	},
 44 	onCreate: function() {
 45 		this.element=$("<"+this.options.tag+"/>").appendTo(this.content);
 46 	},
 47 	display: function(geometry) {
 48 		var sizeRef=Math.min(this.jBlocks.geometry.uWidth,this.jBlocks.geometry.uHeight);
 49 		var marginTop=sizeRef*this.options.emargin[0];
 50 		var marginRight=sizeRef*this.options.emargin[1];
 51 		var marginBottom=sizeRef*this.options.emargin[2];
 52 		var marginLeft=sizeRef*this.options.emargin[3];
 53 		var paddingTop=sizeRef*this.options.epadding[0];
 54 		var paddingRight=sizeRef*this.options.epadding[1];
 55 		var paddingBottom=sizeRef*this.options.epadding[2];
 56 		var paddingLeft=sizeRef*this.options.epadding[3];
 57 		var borderHeight=parseInt(this.element.css("border-top-width"))+parseInt(this.element.css("border-bottom-width"));
 58 		var borderWidth=parseInt(this.element.css("border-left-width"))+parseInt(this.element.css("border-right-width"));
 59 		this.element.css({
 60 			"margin-top": marginTop+"px",
 61 			"margin-right": marginRight+"px",
 62 			"margin-bottom": marginBottom+"px",
 63 			"margin-left": marginLeft+"px",
 64 			"padding-top": paddingTop+"px",
 65 			"padding-right": paddingRight+"px",
 66 			"padding-bottom": paddingBottom+"px",
 67 			"padding-left": paddingLeft+"px",
 68 			"height": this.content.height()-paddingTop-marginTop-paddingBottom-marginBottom-borderHeight,
 69 			"width": this.content.width()-paddingLeft-marginLeft-paddingRight-marginRight-borderWidth,
 70 		});
 71 	},
 72 });
