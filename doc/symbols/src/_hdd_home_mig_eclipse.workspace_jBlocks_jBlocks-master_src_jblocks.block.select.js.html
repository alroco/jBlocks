 12 /**
 13  * <p>An Select block allows the selection of a value amongst a finite set of options. The block has two states: it is initally closed, showing only
 14  * the selected value in a 1 unit high blocks. Once clicked, it displays a regular list.</p>
 15  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 16  * @name Select
 17  * @class Define a Select block to be displayed into a JBlocks.
 18  * @augments List
 19  */
 20 $.jBlocks("defineBlockClass","Select","List",{
 21 	/**
 22 	 * Initializes a Select block.
 23 	 * The method is invoked by the jBlocks core and must never be called directly.
 24 	 * @name Select#init
 25 	 * @function
 26 	 * @param {Object} [layout] see {@link List#init}.
 27 	 * @param {Object} [options] in addition to base class options {@link List#init}:
 28 	 * @param {Boolean} [options.opened=false] the initial state.
 29 	 * @param {String} [options.selected=null] the initial selected value.
 30 	 * @param {String} [options.selWidgetClass] the class to be used for the closed state widget.
 31 	 * @param {Function} [options.onChange] a function(value) being called when the selected value changes.
 32 	 */
 33 	init: function() {
 34 		var $this=this;
 35 		this._super.apply(this,arguments);
 36 		this.options=$.extend({
 37 			opened: false,
 38 			selected: null,
 39 			selWidgetClass: "jblocks-block-grey",
 40 			onChange: function() {},
 41 		},this.options);
 42 		this.superCreateItem=this.options.createItem;
 43 		this.options.createItem=function(item,li) {
 44 			var key=$this.superCreateItem.call(this,item,li);
 45 			li.bind($.jBlocks("click"),function() {
 46 				$this.val(key);
 47 				$this.list.layout.visible=false;
 48 				$this.selWidget.layout.visible=true;
 49 				$this.selfJBlocks.checkLayout();
 50 				$this.options.opened=false;
 51 				$this.jBlocks.checkLayout();
 52 			});
 53 			return key;
 54 		}
 55 		this.superH=this.options.h;
 56 		this.superLayoutH=this.layout.h;
 57 		this.options.h=function(vp) {
 58 			if(!$this.options.opened)
 59 				return 1;
 60 			else if($this.superH)
 61 				return $this.superH.call($this,vp);
 62 			else
 63 				return this.superLayoutH;
 64 		}
 65 	},
 66 	updateSelected: function() {
 67 		var selText="";
 68 		this.ul.find("li").removeClass("jblocks-selected");
 69 		if(this.options.selected!=null && this.items[this.options.selected]) {
 70 			selText=this.items[this.options.selected].li.text();
 71 			this.items[this.options.selected].li.addClass("jblocks-selected");
 72 		}
 73 		this.selWidget.textVal(selText);
 74 	},
 75 	addItem: function(item) {
 76 		this._super(item);
 77 		if(item.value==this.options.selected)
 78 			this.updateSelected();
 79 	},
 80 	onCreate: function() {
 81 		var $this=this;
 82 		this._super.apply(this,arguments);
 83 		this.selWidget=this.selfJBlocks.addNewBlock("Link",{
 84 			h: 1,
 85 			w: $this.layout.w,
 86 			expandHeight: 0,
 87 		},{
 88 			text: "",
 89 			epadding: [0,1,0,0],
 90 			onClick: function() {
 91 				$this.list.layout.visible=true;
 92 				$this.selWidget.layout.visible=false;
 93 				$this.selfJBlocks.checkLayout();
 94 				$this.options.opened=true;
 95 				$this.jBlocks.checkLayout();
 96 			},
 97 		});
 98 		this.selWidget.content.addClass("jblocks-select");
 99 		this.selWidget.anchor.addClass(this.options.selWidgetClass);
100 		if(this.options.opened) {
101 			this.list.layout.visible=true;
102 			this.selWidget.layout.visible=false;
103 		} else {
104 			this.list.layout.visible=false;
105 			this.selWidget.layout.visible=true;			
106 		}
107 		this.updateSelected();
108 	},
109 	/**
110 	 * Get or set the selection.
111 	 * @name Select#val
112 	 * @function
113 	 * @param {String} [text] if an argument is used, the current selection is replaced.
114 	 * @returns the currently selected key.
115 	 */
116 	val: function() {
117 		if(arguments.length>0) {
118 			var selected=arguments[0];
119 			if(selected!=this.options.selected) {
120 				this.options.selected=selected;
121 				this.updateSelected();
122 				this.options.onChange.call(this,selected);
123 			}
124 		}
125 		return this.options.selected;
126 	},
127 });
