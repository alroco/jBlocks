 12 /**
 13  * <p>A Foo block just displays a letter. It is mainly used for testing.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Foo
 16  * @class Define a Foo block to be displayed into a JBlocks.
 17  * @augments Block
 18  */
 19 $.jBlocks("defineBlockClass","Foo","Block",{
 20 	/**
 21 	 * Initializes a Foo block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Foo#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}
 26 	 * @param {Object} [options] see {@link Block#init}
 27 	 */
 28 	init: function() {
 29 		this._super.apply(this,arguments);
 30 		this.letter=String.fromCharCode(65+(this.constructor.fooIndex++));
 31 	},
 32 	onCreate: function() {
 33 		this.content.css({
 34 			"background-color": "rgba("+Math.floor(Math.random()*256)+","+Math.floor(Math.random()*256)+","+Math.floor(Math.random()*256)+",0.5)",
 35 			"text-align": "center",
 36 		}).text(this.letter);
 37 		this._super.call(this);
 38 	},
 39 	display: function(geometry) {
 40 		//JocLog("display",this.id,geometry);
 41 		this.content.css({
 42 			"line-height": geometry.height+"px",
 43 			"font-size": Math.min(geometry.height/2,geometry.width*0.8)+"pt",			
 44 		});
 45 	},
 46 }).fooIndex=0;
