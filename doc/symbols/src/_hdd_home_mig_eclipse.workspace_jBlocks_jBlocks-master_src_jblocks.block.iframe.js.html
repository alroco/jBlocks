 12 /**
 13  * <p>An IFrame block displays an HTML iframe as part of the block content.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name IFrame
 16  * @class Define an IFrame block to be displayed into a JBlocks.
 17  * @augments Icon
 18  */
 19 $.jBlocks("defineBlockClass","IFrame","Block",{
 20 	/**
 21 	 * Initializes an IFrame block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name IFrame#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {Array} [options.src] the source URL for the iframe.
 28 	 * @param {Object} [options.data] optional data that can be passed to opened page if it is supporting jBlocks. See {@link JBlocks.getFrameData}
 29 	 * for how to get the passed data.
 30 	 * @param {Boolean} [options.syncViewportSize=false] if set to true and the opened URL is a jBlocks-managed page, this jBlocks viewport size will
 31 	 * match the block size.
 32 	 * 
 33 	 */
 34 	init: function() {
 35 		this._super.apply(this,arguments);
 36 		this.options=$.extend({
 37 			src: "about:blank",
 38 			data: null,
 39 			syncViewportSize: false,
 40 		},this.options);
 41 	},
 42 	onCreate: function() {
 43 		var $this=this;
 44 		this.frame=$("<iframe/>")
 45 			.attr("frameborder","0")
 46 			.attr("scrolling","no")
 47 			.attr("marginwidth","0")
 48 			.attr("marginheight","0")
 49 			.attr("allowTransparency",true)
 50 			.attr("webkitallowfullscreen",true)
 51 			.attr("mozallowfullscreen",true)
 52 			.attr("allowfullscreen",true)
 53 			.css({
 54 				"background-color": "transparent",
 55 			});
 56 		if(this.options.syncViewportSize) {
 57 			if(!this.options.data)
 58 				this.options.data={}
 59 			$.extend(this.options.data,{
 60 				onResize: function(geometry,viewport) {
 61 					if(!$this.jBlocks.canFullscreen() || !document[$this.jBlocks.fullscreenMethods.element]) {
 62 						viewport.w=$this.actualLayout.w;
 63 						viewport.h=$this.actualLayout.h;
 64 					} else
 65 						this.options.defaultOnResize.call(this,geometry,viewport);
 66 				}				
 67 			});
 68 		}
 69 		if(this.options.data) {
 70 			if(typeof $.jBlocks.frameData=="undefined")
 71 				$.jBlocks.frameData={
 72 					index: 0,
 73 					data: this.options.data,
 74 				}
 75 			var frameData=$.jBlocks.frameData;
 76 			var index=frameData.index++;
 77 			frameData.data[index]=this.options.data;	
 78 			this.frame.attr("name","jblocks-data-"+index);
 79 		}
 80 		this.frame.appendTo(this.content);
 81 	},
 82 	display: function(geometry) {
 83 		this.frame.attr("width",geometry.width).attr("height",geometry.height);
 84 		if(geometry.width>0 && geometry.height>0 && this.frame.attr("src")===undefined)
 85 			this.frame.attr("src",this.options.src);
 86 	}
 87 });
