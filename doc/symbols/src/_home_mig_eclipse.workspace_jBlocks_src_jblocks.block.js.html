 12 /**
 13  * <p>A Block represents a rectangular area inside a JBlocks viewport. The exact position of the Block depends on the result of the layout calculation
 14  * performed by the library based on the created Block objects inside the JBlocks.</p>
 15  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 16  * @name Block
 17  * @class Define a block to be displayed into a JBlocks.
 18  */
 19 $.jBlocks("defineBlockClass","Block","Class",{
 20 	/**
 21 	 * Initializes a block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly. 
 23 	 *
 24 	 * @name Block#init
 25 	 * @function
 26 	 * @param {Object} [layout] Object defining the layout preferences for the Block.
 27 	 * @param {Number} [layout.priority=0] When performing a layout, the actual Block objects positions are calculated starting with highest priority. 
 28 	 * @param {Number} [layout.importance=0] If the JBlocks has option <i>fixed</i> set to <i>both</i> and there is no space enough to display all the Blocks,
 29 	 * the <i>importance</i> parameter is used to determine which blocks will be hidden. 
 30 	 * @param {Integer} [layout.w=1] the minimum virtual width required for this block.
 31 	 * @param {Integer} [layout.h=1] the minimum virtual height required for this block.
 32 	 * @param {String} [layout.pref="default"] the preference for positioning the block:<ul>
 33 	 * <li><i>default</i>: if JBlocks <i>fixed</i> option is set to <i>height</i>, the <i>pref</i> field behaves as if it was "left top" otherwise as "top left".</li>
 34 	 * <li><i>top left</i>: displays the block as top as possible then as left as possible.</li>
 35 	 * <li><i>left top</i>: displays the block as left as possible then as top as possible.</li>
 36 	 * <li><i>top right</i>: displays the block as top as possible then as right as possible in the viewable viewport area.</li>
 37 	 * <li><i>right top</i>: displays the block as right as possible in the viewable area then as top as possible.</li>
 38 	 * <li><i>bottom left</i>: displays the block as bottom as possible in the viewable area then as left as possible.</li>
 39 	 * <li><i>left bottom</i>: displays the block as left as possible then as bottom as possible in the viewable area.</li>
 40 	 * <li><i>bottom right</i>: displays the block as bottom as possible then as right as possible, in the viewable viewport area.</li>
 41 	 * <li><i>right bottom</i>: displays the block as right as possible then as bottom as possible, in the viewable area.</li>
 42 	 * </ul>
 43 	 * @param {Boolean} [layout.visible=true] should this block be visible.
 44 	 * @param {Number} [layout.expandWidth=1] if set to 0, the block will never expand and will keep the width given by the <i>w</i> parameter. Otherwise,
 45 	 * if JBlock option <i>fillHoles</i> is set to true, the layout algorithm tries to expand the neighbor blocks to fill the holes. Blocks with highest 
 46 	 * <i>expandWidth</i> are expanded in priority.
 47 	 * @param {Number} [layout.expandHeight=1] if set to 0, the block will never expand and will keep the height given by the <i>h</i> parameter. Otherwise,
 48 	 * if JBlock option <i>fillHoles</i> is set to true, the layout algorithm tries to expand the neighbor blocks to fill the holes. Blocks with highest 
 49 	 * <i>expandHeight</i> are expanded in priority.
 50 	 * 
 51 	 * @param {Object} [options] Object defining the options for the Block.
 52 	 * @param {String} [options.id] a string to be used as block id. Only letters, digits and dash characters are to be used. If not set, a safe value is automatically used. 
 53 	 * @param {String} [options.blockClass=null] an additional class to be added to the block. For instance, the block may define a border.
 54 	 * @param {String} [options.blockContentClass=null] an additional class to be added to the block content area. For instance, the block may define a background-color.
 55 	 * @param {NumberArray} [options.margin] the margins [top,right,bottom,left] expressed relatively to the block size. If not set, the containing JBlocks margin applies.
 56 	 * @param {NumberArray} [options.padding] the paddings [top,right,bottom,left] expressed relatively to the block size. If not set, the containing JBlocks margin applies.
 57 	 * @param {Number} [options.borderRadius=0.1] the block border radius expressed relatively to the block size.
 58 	 * @param {Number} [options.contentBorderRadius=0.1] the block content border radius expressed relatively to the block content size.
 59 	 * @param {Boolean} [options.requireAuth=false] if set to true, the becomes visible only when the user is logged in.
 60 	 * @param {Boolean} [options.requireAnon=false] if set to true, the becomes visible only when the user is logged out.
 61 	 * @param {Number} [options.resizeAnimateDelay=400] the number of milliseconds to animate the block resizing.
 62 	 * @param {Function} [options.onCreate] function called when the block is created.
 63 	 * @param {Function} [options.onRemove] function called when the block is removed.
 64 	 * @param {Function} [options.onResize] function(geometry) called when the block has resized.
 65 	 * @param {Function} [options.onStartResize] function(geometry) called when the block starts resizing. Parameter <i>geometry</i> represents the final size.
 66 	 * @param {Function} [options.w] function(viewport) called before a relayout so the block can returns its virtual width overloading the provided <i>layout.w</i>.
 67 	 * @param {Function} [options.h] function(viewport) called before a relayout so the block can returns its virtual height overloading the provided <i>layout.h</i>.
 68 	 * @param {Array} [options.layouts] an array containing the various layouts a block can have. The Block API provides methods to navigate amongst the 
 69 	 * defined layouts, allowing blocks to grow or shrink. 
 70 	 * Each layout entry contains:<ul>
 71 	 * <li><i>w</i>: an integer or a function(viewport) returning the block width.</li>
 72 	 * <li><i>h</i>: an integer or a function(viewport) returning the block height.</li>
 73 	 * </ul>
 74 	 * @param {Integer} [options.layoutIndex] the initial index within the <i>layouts</i> array.
 75 	 * @param {String} [options.tooltip] if set, the given string appears as a tooltip when the mouse is over the block.
 76 	 * @param {Boolean} [options.earlyLayout=true] during a single layout process, a block may be asked to resize several times. If resizing is expensive,
 77 	 * the <i>earlyLayout</i> option can be set to <i>false</i> so the block is resized only once at the end of the layout operation.
 78 	 * @param {Float} [options.shadow] defines a shadow for the block (i.e. 0.15). If unset, the JBlocks <i>blockShadow</i> option applies. 
 79 	 * 
 80 	 */
 81 	init: function(layout,options) {
 82 		var $this=this;
 83 		if(arguments.length<2)
 84 			options={};

 86 			layout={};

 89 		else

 91 		this.layout={
 92 			priority: 0,
 93 			importance: 0,
 94 			w: 1,
 95 			h: 1,
 96 			pref: "default",
 97 			visible: true,
 98 			expandWidth: 1,
 99 			expandHeight: 1,
100 		}
101 		this.actualLayout={ w:1, h: 1 };
102 		$.extend(this.layout,layout);
103 		this.maximized=false;
104 		/**
105 		 * The JBlocks object managing the block.
106 		 * @name Block#jBlocks
107 		 * @field
108 		 */
109 		this.jBlocks=null;
110 		/**
111 		 * The name of the class for this block object.
112 		 * @name Block#className
113 		 * @field
114 		 */
115 		this.className="Block";
116 		this.options={
117 			blockClass: null,
118 			blockContentClass: null,
119 			margin: null,
120 			padding: null,
121 			borderRadius: 0.15,
122 			contentBorderRadius: 0.0,
123 			capture: false,
124 			maximizable: true,
125 			requireAuth: false,
126 			requireAnon: false,
127 			autoFit: false,
128 			resizeAnimateDelay: 400,
129 			debugBlockLayout: false,
130 			onCreate: function(){},
131 			onRemove: function(){},
132 			onResize: function(){},
133 			onStartResize: function(){},
134 			tooltip: null,
135 			earlyLayout: true,
136 		}
137 		this.tags={};
138 		$.extend(this.options,options);
139 		if(this.options.requireAuth && $.jBlocks.loggedIn==null)
140 			this.layout.visible=false;
141 		if(this.options.requireAnon && $.jBlocks.loggedIn!=null)
142 			this.layout.visible=false;
143 	},
144 	create: function(anchor) {
145 		//JocLog("create",this);
146 
147 		if(this.options.layouts) {
148 			if(typeof this.options.layoutIndex=="undefined")
149 				this.options.layoutIndex=Math.floor((this.options.layouts.length-1)/2);
150 			this.options.w=function(viewport) {
151 				var layout=$this.options.layouts[$this.options.layoutIndex];
152 				switch(typeof layout.w) {
153 				case 'number':
154 					return layout.w;
155 				case 'function':
156 					return layout.w.call($this,viewport);
157 				default: 
158 					return $this.layout.w;
159 				}
160 			}
161 			this.options.h=function(viewport) {
162 				var layout=$this.options.layouts[$this.options.layoutIndex];
163 				switch(typeof layout.h) {
164 				case 'number':
165 					return layout.h;
166 				case 'function':
167 					return layout.h.call($this,viewport);
168 				default: 
169 					return $this.layout.h;
170 				}
171 			}
172 		}
173 
174 		var $this=this;
175 		var blockClass=this.options.blockClass || this.jBlocks.options.blockClass;
176 		this.anchor=$("<div/>").appendTo(anchor.children(".jblocks-frame").children(".jblocks-frame-content"));
177 		this.anchor.addClass("jblocks-block")
178 			.addClass("jblocks-"+this.jBlocks.jBlocksIndex)
179 			.attr("id","jblocks-block-"+this.jBlocks.jBlocksIndex+"-"+this.id)
180 			.css({
181 				top: $("body").height()/2,
182 				left: $("body").width()/2,
183 				width: 1,
184 				height: 1,
185 			});
186 		if(blockClass)
187 			this.anchor.addClass(blockClass);
188 		if(this.options.debugBlockLayout || this.jBlocks.options.debugBlockLayout)
189 			this.anchor.bind("mouseover",function(event) {
190 				$("head title").text($this.className+" "+JSON.stringify($this.layout));
191 				event.stopPropagation();
192 			});
193 
194 		this.captMask=$("<div/>").addClass("jblocks-block-mask").appendTo(this.anchor);
195 		var blockContentClass=this.options.blockContentClass || this.jBlocks.options.blockContentClass;
196 		this.content=$("<div/>").addClass("jblocks-block-content").appendTo(this.anchor);
197 		if(blockContentClass)
198 			this.content.addClass(blockContentClass);
199 		if(this.options.tooltip)
200 			this.content.attr("title",this.options.tooltip);
201 		this.onCreate();
202 		this.options.onCreate.call(this);
203 	},

222 	/**
223 	 * Returns whether the block uses a set of layouts from <i>options.layouts</i> and can step up to a bigger size.

225 	 * @function
226 	 * @returns whether calling the {@link Block#grow} method will have an effect on the block size.

230 	},
231 	/**
232 	 * Step up to the higher layout size if a set of layouts from <i>options.layouts</i> has been defined.

235 	 * @see Block#canGrow
236 	 */

238 		if(this.canGrow()) {
239 			this.options.layoutIndex++;

241 			this.jBlocks.checkLayout();
242 			this.focus();
243 		}
244 	},
245 	/**
246 	 * Returns whether the block uses a set of layouts from <i>options.layouts</i> and can step down to a smaller size.

249 	 * @returns whether calling the {@link Block#shrink} method will have an effect on the block size. 
250 	 */

261 		if(this.canShrink()) {
262 			this.options.layoutIndex--;

264 			this.jBlocks.checkLayout();
265 			this.focus();
266 		}

325 	/**
326 	 * Called just before the block is about to be resized. 
327 	 * The method is invoked by the jBlocks core and must never be called directly.
328 	 * @name Block#onStartResize
329 	 * @function
330 	 * @param {Object} geometry the new block content geometry.
331 	 * @param {Number} geometry.width the width of the block content in pixels
332 	 * @param {Number} geometry.height the height of the block content in pixels

337 	getContentGeometry: function(geometry,uWidth,uHeight) {
338 		var sizeRef=Math.min(uWidth,uHeight);
339 		var margin=this.jBlocks.options.margin;
340 		if(this.options.margin)
341 			margin=this.options.margin;
342 		var padding=this.jBlocks.options.padding;
343 		if(this.options.padding)
344 			padding=this.options.padding;
345 		var spacing={
346 				"margin-top": sizeRef*margin[0],
347 				"margin-bottom": sizeRef*margin[2],
348 				"margin-left": sizeRef*margin[3],
349 				"margin-right": sizeRef*margin[1],
350 				"padding-top": sizeRef*padding[0],
351 				"padding-bottom": sizeRef*padding[2],
352 				"padding-left": sizeRef*padding[3],
353 				"padding-right": sizeRef*padding[1],
354 				"border-left": parseInt(this.anchor.css("border-left-width")),
355 				"border-right": parseInt(this.anchor.css("border-right-width")),
356 				"border-top": parseInt(this.anchor.css("border-top-width")),
357 				"border-bottom": parseInt(this.anchor.css("border-bottom-width")), 
358 		}
359 		var contentGeo={
360 			spacing: spacing,
361 			sizeRef: sizeRef,
362 			width: geometry.width
363 				-(margin[1]+margin[3])*sizeRef
364 				-(padding[1]+padding[3])*sizeRef
365 				-spacing["border-left"]
366 				-spacing["border-right"],
367 			height: geometry.height
368 				-(margin[0]+margin[2])*sizeRef
369 				-(padding[0]+padding[2])*sizeRef
370 				-spacing["border-right"]
371 				-spacing["border-bottom"],
372 		};
373 		return contentGeo;
374 	},
375 	/**
376 	 * Called just after the block has been resized. 
377 	 * The method is invoked by the jBlocks core and must never be called directly.
378 	 * @name Block#onResize
379 	 * @function
380 	 * @param {Object} geometry the new block content geometry.
381 	 * @param {Number} geometry.width the width of the block content in pixels
382 	 * @param {Number} geometry.height the height of the block content in pixels
383 	 */
384 	onResize: function(geometry) {
385 		var contentGeo=this.getContentGeometry(geometry,this.jBlocks.geometry.uWidth,this.jBlocks.geometry.uHeight);
386 
387 		//$.jBlocks("log","contentGeo",geometry);
388 		this.content.width(contentGeo.width);
389 		this.content.height(contentGeo.height);
390 		this.content.css("border-radius",
391 				contentGeo.sizeRef*this.options.contentBorderRadius+"px");
392 		this.anchor.css({
393 			"margin-top": contentGeo.spacing["margin-top"]+"px",
394 			"margin-bottom": contentGeo.spacing["margin-bottom"]+"px",
395 			"margin-left": contentGeo.spacing["margin-left"]+"px",
396 			"margin-right": contentGeo.spacing["margin-right"]+"px",
397 			"padding-top": contentGeo.spacing["padding-top"]+"px",
398 			"padding-bottom": contentGeo.spacing["padding-bottom"]+"px",
399 			"padding-left": contentGeo.spacing["padding-left"]+"px",
400 			"padding-right": contentGeo.spacing["padding-right"]+"px",
401 		});
402 		this.anchor.width(contentGeo.width);
403 		this.anchor.height(contentGeo.height);
404 		this.anchor.css("border-radius",
405 				contentGeo.sizeRef*this.options.borderRadius+"px");
406 		this.captMask.width(this.anchor.outerWidth(false)
407 				-contentGeo.spacing["border-left"]-contentGeo.spacing["border-right"]);
408 		this.captMask.height(this.anchor.outerHeight(false)
409 				-contentGeo.spacing["border-top"]-contentGeo.spacing["border-bottom"]);
410 		if(this.options.capture)
411 			this.captMask.show();
412 		else
413 			this.captMask.hide();
414 		this.options.onResize.call(this,contentGeo);
415 		var blockShadow=this.jBlocks.options.blockShadow;
416 		if(typeof this.options.shadow!="undefined")
417 			blockShadow=this.options.shadow;
418 		if(blockShadow!=0) {
419 			var shadowW=this.jBlocks.geometry.uWidth*blockShadow;
420 			var shadowH=this.jBlocks.geometry.uHeight*blockShadow;
421 			var blur=contentGeo.sizeRef*blockShadow;
422 			this.anchor.css({
423 				"-webkit-box-shadow": "-"+shadowW+"px "+shadowH+"px "+(blur*2.5)+"px rgba(43, 45, 50, 0.55)",
424 				"box-shadow": "-"+shadowW+"px "+shadowH+"px "+(blur*2.5)+"px rgba(43, 45, 50, 0.55)",
425 			});
426 		}
427 		this.display(contentGeo);
428 		/*
429 		if(this.options.autoFit)
430 			this.fit();
431 		*/
432 	},
