 12 /**
 13  * <p>A Box block holds a header with title and menu. The class must be overloaded to add content.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Box
 16  * @class Define a Box block to be displayed into a JBlocks.
 17  * @augments JBlock
 18  */
 19 $.jBlocks("defineBlockClass","Box","JBlock",{
 20 	/**
 21 	 * Initializes a Box block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Box#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link JBlock#init}:
 27 	 * @param {Array} [options.title] the title to appear in the Box header.
 28 	 * @param {Boolean} [options.haveClose=false] have an icon to close the block.
 29 	 * @param {Boolean} [options.haveFullscreen=false] have a Fullscreen entry in the menu.
 30 	 * @param {Boolean} [options.haveSearch=false] have a icon to search.
 31 	 * @param {Array} [options.menuItems] additional menu items.
 32 	 * @param {Function} [options.onSearchClick] additional function to be called when the search icon is clicked.
 33 	 * @param {String} [options.closeIconClass="jblocks-icon-close"] class to be used for Close icon.
 34 	 * @param {String} [options.shrinkIconClass="jblocks-icon-shrink"] class to be used for Shrink icon.
 35 	 * @param {String} [options.growIconClass="jblocks-icon-grow"] class to be used for Grow icon.
 36 	 * @param {String} [options.searchIconClass="jblocks-icon-search"] class to be used for Search icon.
 37 	 * @param {String} [options.iconsBaseClass="jblocks-block-icon"] base class to be used for icons.
 38 	 * @param {String} [options.focusOnTitleClick=true] focus block if title is clicked
 39 	 */
 40 	init: function() {
 41 		this._super.apply(this,arguments);
 42 		this.options=$.extend({
 43 			title: "",
 44 			focusOnTitleClick: true,
 45 			haveClose: false,
 46 			haveSearch: false,
 47 			haveMaximize: true,
 48 			haveFullscreen: false,
 49 			closeIconClass:"jblocks-icon-close",
 50 			shrinkIconClass:"jblocks-icon-shrink",
 51 			growIconClass:"jblocks-icon-grow",
 52 			searchIconClass:"jblocks-icon-search",
 53 			iconsBaseClass:"jblocks-block-icon",
 54 			menuItems: [],
 55 			onSearchClick: function() {},
 56 		},this.options);
 57 	},
 58 	onCreate: function() {
 59 		var $this=this;
 60 		this._super();
 61 		var _onResize=this.selfJBlocks.options.onResize;
 62 		this.selfJBlocks.options.onResize=function() {
 63 			_onResize.apply(this,arguments);
 64 			if($this.options.layouts) {
 65 				if($this.canGrow())
 66 					$this.growIcon.layout.visible=true;
 67 				else
 68 					$this.growIcon.layout.visible=false;
 69 				if($this.canShrink())
 70 					$this.shrinkIcon.layout.visible=true;
 71 				else
 72 					$this.shrinkIcon.layout.visible=false;
 73 			}
 74 		}
 75 		var menuItems=[];
 76 		if(this.options.haveFullscreen)
 77 			menuItems.push({
 78             	label: this.t("fullscreen"),
 79             	onClick: function() {
 80             		this.layout.visible=false;
 81             		$this.fullscreen($this.selfJBlocks.anchor,function(entering) {
 82             			if(entering)
 83             				$this.maximize();
 84             			else
 85             				$this.unMaximize();
 86         				$(window).trigger('resize');
 87             		});
 88             	},
 89             	shouldDisplay: function() {
 90             		return $this.canFullscreen();
 91             	},
 92 			});
 93 		for(var i in this.options.menuItems)
 94 			menuItems.push(this.options.menuItems[i]);
 95 		
 96 		if(menuItems.length>0) 
 97 			this.selfJBlocks.addNewBlock("MenuIcon",{
 98 					priority:1003,
 99 					pref:"top left",
100 					h: 1,
101 					w: 1,
102 					expandHeight: 0,
103 				},{
104 					onClick: function(){alert("Open menu");},
105 					menuItems: menuItems,
106 				});
107 		if(this.options.haveClose){
108 			this.selfJBlocks.addNewBlock("Icon",{
109 					priority:1002,
110 					pref:"top left",
111 				},{
112 					iconClass: this.options.iconsBaseClass,
113 					blockContentClass: this.options.closeIconClass,
114 					onClick: function() {
115 						$this.unMaximize();
116 						$this.jBlocks.removeBlock($this.id);
117 					},
118 					tooltip: this.t('close'),
119 				});
120 		}
121 		if(this.options.haveSearch)
122 			this.selfJBlocks.addNewBlock("Icon",{
123 					priority:1001,
124 					pref:"top right",
125 				},{
126 					iconClass: this.options.iconsBaseClass,
127 					blockContentClass: this.options.searchIconClass,
128 					onClick: function() {
129 						$this.options.onSearchClick.apply($this);
130 					},
131 					tooltip: this.t('search'),
132 				});
133 		if(this.options.layouts) {
134 			this.growIcon=this.selfJBlocks.addNewBlock("Icon",{
135 				priority:1001,
136 				pref:"top right",
137 			},{
138 				iconClass: this.options.iconsBaseClass,
139 				blockContentClass: this.options.growIconClass,
140 				onClick: function() {
141 					$this.grow();
142 				},
143 				tooltip: this.t('grow'),
144 			});
145 			this.shrinkIcon=this.selfJBlocks.addNewBlock("Icon",{
146 				priority:1001,
147 				pref:"top right",
148 			},{
149 				iconClass: this.options.iconsBaseClass,
150 				blockContentClass: this.options.shrinkIconClass,
151 				onClick: function() {
152 					$this.shrink();
153 				},
154 				tooltip: this.t('shrink'),
155 			});
156 		}
157         var txt=this.selfJBlocks.addNewBlock("Text",{
158     			h: 1,
159     			w: 2,
160     			priority: 1000,
161     			expandWidth: 10,
162     			expandHeight: 0,
163         	},{
164         		containerTag: "h1",
165         		text: this.options.title,
166         	});
167         if (this.options.focusOnTitleClick){
168         	txt.content.bind('click',function(){
169         		$this.focus();
170         		$this.jBlocks.checkLayout();
171         	});
172         }
173 	},
174 	/**
175 	 * This method is to be used when the Box block is intended to have a single block as content. A new block is created, eventually replacing a former content block.
176 	 * @name Box#setContent
177 	 * @function
178 	 * @param {String} className the name of the class for the content block.
179 	 * @param {Object} [options] options for the new block. The options parameters depends on the actual block class.
180 	 * @returns the newly created block.
181 	 */
182 	setContent: function(className,options) {
183 		if(arguments.length<2)
184 			options={};
185 		for(var bi in this.selfJBlocks.blocks) {
186 			var block=this.selfJBlocks.blocks[bi];
187 			if(block._isBoxContent)
188 				this.selfJBlocks.removeBlock(block.id);
189 		}
190         var newBlock=this.selfJBlocks.addNewBlock(className,{
191     		w: 1,
192     		h: 2,
193     		pref: "right bottom",
194     		priority: 999,
195     		expandWidth: 2,
196     		expandHeight: 2,
197         },$.extend(options,{
198         	boxBlock: this,
199         }));
200         newBlock._isBoxContent=true;
201         return newBlock;
202 	},
203 });
