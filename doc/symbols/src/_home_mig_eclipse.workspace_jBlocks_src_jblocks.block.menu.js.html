 12 /**
 13  * <p>A Menu block holds a set of links.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Menu
 16  * @class Define a Menu block to be displayed into a JBlocks.
 17  * @augments JBlock
 18  */
 19 $.jBlocks("defineBlockClass","Menu","JBlock",{
 20 	/**
 21 	 * Initializes a Menu block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Menu#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {Array} [options.menuItems] menu items associated with the Menu block. Each menu item contains the fields:<ul>
 28 	 * <li><i>label</i>: the text to be displayed.</li>
 29 	 * <li><i>onClick</i>: a function to be called when the menu item is clicked.</li>
 30 	 * <li><i>shouldDisplay</i>: a function to be called when the menu item is about to be displayed. If it returns false, the entry is not shown.</li>
 31 	 * </ul>
 32 	 */
 33 	init: function(block,data) {
 34 		this._super.apply(this,arguments);
 35 		$.extend(this.options,{
 36 			blockContentClass: "jblocks-block-menu",
 37 		});
 38 		this.options=$.extend({
 39 			w: 1,
 40 			h: 1,
 41 			menuItems: [],
 42 		},this.options);
 43 	},
 44 	onCreate: function() {
 45 		var $this=this;
 46 		this._super();
 47 		this.selfJBlocks.viewport.fixed="width";
 48 		this.selfJBlocks.options.scrollable=true;
 49 		//this.selfJBlocks.anchor.addClass("jblocks-block-menu");
 50 		this.layout.h=this.jBlocks.viewport.h-1;
 51 		this.layout.w=this.jBlocks.viewport.w;
 52 		this.menuItems=[];
 53 		for(var i in this.options.menuItems) {
 54 			var menuItem={
 55 				label: "--",
 56 				onClick: function() {},
 57 				shouldDisplay: function() { return true; },
 58 			}
 59 			$.extend(menuItem,this.options.menuItems[i]);
 60 			var blockItem=this.selfJBlocks.addNewBlock("Link",{
 61 					w:1,
 62 					h:1,
 63 					pref:"left top",
 64 					expandHeight:0,
 65 					expandWidth:0,
 66 				},{
 67 					text:menuItem.label,
 68 					containerTag: "a",
 69 				});
 70 			this.menuItems.push({
 71 				data: menuItem,
 72 				block: blockItem,
 73 			});
 74 			(function(menuItem) {
 75 				blockItem.anchor.bind("click",function() {
 76 					menuItem.onClick.call($this);
 77 				});
 78 			})(menuItem); 
 79 		}
 80 		this.selfJBlocks.updateOptions({
 81 			onResize: function(geometry,viewport) {
 82 				viewport.w=$this.options.w;
 83 			},			
 84 		});
 85 	},
 86 	display: function(geometry) {
 87 		this._super();
 88 		var itemsCount=0;
 89 		for(var i in this.menuItems) {
 90 			var menuItem=this.menuItems[i];
 91 			if(menuItem.data.shouldDisplay()) {
 92 				menuItem.block.layout.visible=true;
 93 				itemsCount++;
 94 			} else 
 95 				menuItem.block.layout.visible=false;
 96 		}
 97 		var h=Math.min(itemsCount,this.jBlocks.viewport.h-1);
 98 		this.layout.h=h;
 99 		this.layout.w=4;
100 		this.jBlocks.checkLayout();
101 		/*
102 		this.selfJBlocks.updateOptions({
103 			scrollable: true,
104 		});
105 		*/
106 		this.selfJBlocks.updateViewport({
107 			h: h,
108 		});
109 		//alert(itemsCount+" "+JSON.stringify(this.selfJBlocks.viewport));
110 		/*
111 		if(itemsCount!=this.selfJBlocks.viewport) {
112 			this.selfJBlocks.updateViewport({
113 				h: itemsCount,
114 			});
115 		} else
116 			this.content.trigger("refresh");
117 		*/
118 		/*
119 		var lineHeight=Math.max(geometry.uHeight,20);
120 		var h=3*lineHeight;
121 		*/
122 		//$.jBlocks("log","display JBlock",this.selfJBlocks.viewport,this.selfJBlocks.geometry);
123 		//this.content.text();
124 	},
125 });
