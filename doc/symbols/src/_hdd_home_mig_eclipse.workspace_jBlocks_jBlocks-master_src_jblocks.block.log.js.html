 12 /**
 13  * <p>A Log block displays the jBlocks block. It is mainly used for debugging.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Log
 16  * @class Define a Log block to be displayed into a JBlocks.
 17  * @augments ScrollableHtml
 18  */
 19 $.jBlocks("defineBlockClass","Log","ScrollableHtml",{
 20 	/**
 21 	 * Initializes a Log block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Log#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link ScrollableHtml#init}
 26 	 * @param {Object} [options] in addition to base class options {@link ScrollableHtml#init}:
 27 	 * @param {Number} [options.refreshPeriod=500] the number of millisecond to check and refresh the scrollbar. 
 28 	 * @param {Number} [options.maxEntries=100] the maximum number of lines in the log. 
 29 	 */
 30 	init: function() {
 31 		this._super.apply(this,arguments);
 32 		this.options.padding=[0.15,0.15,0.15,0.15];
 33 		this.nbEntries=0;
 34 		this.options=$.extend({
 35 			refreshPeriod: 500,
 36 			maxEntries: 100,
 37 		},this.options);
 38 	},
 39 	onCreate: function() {
 40 		var $this=this;
 41 		this._super.call(this);
 42 		this.getHtmlContainer().addClass("jblocks-log").css({
 43 			"background-color": "White",
 44 			"border": "2px inset #808080",
 45 			"padding": 5,
 46 		});
 47 		this.updateTimer=setInterval(function() {
 48 			$this.updateLog();
 49 		},this.options.refreshPeriod);
 50 	},
 51 	onRemove: function() {
 52 		if(this.updateTimer)
 53 			clearInterval(this.updateTimer);
 54 	},
 55 	updateLog: function() {
 56 		var nbEntries=this.getHtmlContainer().children().length;
 57 		if(nbEntries!=this.nbEntries) {
 58 			var excess=nbEntries-this.options.maxEntries;
 59 			if(excess>0)
 60 				this.getHtmlContainer().children().slice(0,excess).remove();
 61 			this.checkState();
 62 			this.scrollBottom();
 63 			this.nbEntries=this.getHtmlContainer().children().length;
 64 		}
 65 	},
 66 	display: function() {
 67 		this._super();
 68 		this.scrollBottom();
 69 	},
 70 	/**
 71 	 * Clears the log.
 72 	 * @name Log#clearLog
 73 	 * @function
 74 	 */
 75 	clearLog: function() {
 76 		this.getHtmlContainer().empty();
 77 		this.checkState();
 78 	},
 79 });

 81 /**
 82  * <p>A LogBox block wraps a Log block into a box, so it can be maximized or closed. It is mainly used for debugging.</p>
 83  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 84  * @name LogBox
 85  * @class Define a LogBox block to be displayed into a JBlocks.
 86  * @augments Box
 87  * @see Log
 88  */
 89 $.jBlocks("defineBlockClass","LogBox","Box",{
 90 	/**
 91 	 * Initializes a LogBox block.
 92 	 * The method is invoked by the jBlocks core and must never be called directly.
 93 	 * @name LogBox#init
 94 	 * @function
 95 	 * @param {Object} [layout] see {@link Box#init}
 96 	 * @param {Object} [options] see {@link Box#init}
 97 	 */
 98 	init: function() {
 99 		var $this=this;
100 		this._super.apply(this,arguments);
101 		this.options.title=this.t("Log");
102 		this.options.jBlocksBlockClass="jblocks-block-brown";
103 		this.options.haveClose=true;
104 		this.options.menuItems.push({
105 			label: "Clear",
106 			onClick: function() {
107 				this.layout.visible=false;
108 				$this.clearLog();
109 			},
110 		});
111 	},
112 	onCreate: function() {
113 		this._super();
114         this.logPanel=this.setContent("Log");
115 	},
116 	/**
117 	 * Clears the log.
118 	 * @name LogBox#clearLog
119 	 * @function
120 	 */
121 	clearLog: function() {
122 		this.logPanel.clearLog();
123 	},
124 });
