 91 					var point2=points[pi2];

172 					blockPref="left top";
173 				else

175 			var suitableSortFn=null;
176 			function PrefTop(z1,z2) {
177 				return z1.r-z2.r;

179 			function PrefLeft(z1,z2) {

190 				return Math.abs(z1.c+z1.w-viewport.w)-Math.abs(z2.c+z2.w-viewport.w);

194 			}

196 				case 'top right': suitableSortFn=function(z1,z2) {
197 					return PrefTop(z1,z2) || PrefRight(z1,z2) || PrefArea(z1,z2);
198 				};
199 				break;
200 				case 'right top': suitableSortFn=function(z1,z2) {
201 					return PrefRight(z1,z2) || PrefTop(z1,z2) || PrefArea(z1,z2);
202 				};
203 				break;
204 				case 'bottom left': suitableSortFn=function(z1,z2) {
205 					return PrefBottom(z1,z2) || PrefLeft(z1,z2) || PrefArea(z1,z2);
206 				};
207 				break;
208 				case 'left bottom': suitableSortFn=function(z1,z2) {
209 					return PrefLeft(z1,z2) || PrefBottom(z1,z2) || PrefArea(z1,z2);
210 				};
211 				break;
212 				case 'bottom right': suitableSortFn=function(z1,z2) {
213 					return PrefBottom(z1,z2) || PrefRight(z1,z2) || PrefArea(z1,z2);
214 				};
215 				break;
216 				case 'right bottom': suitableSortFn=function(z1,z2) {
217 					return PrefRight(z1,z2) || PrefBottom(z1,z2) || PrefArea(z1,z2);
218 				};
219 				break;
220 				case 'left top': suitableSortFn=function(z1,z2) {
221 					return PrefLeft(z1,z2) || PrefTop(z1,z2) || PrefArea(z1,z2);
222 				};
223 				break;
224 				case 'top left': suitableSortFn=function(z1,z2) {
225 					return PrefTop(z1,z2) || PrefLeft(z1,z2) || PrefArea(z1,z2);
226 				};
227 				break;
228 				case 'center': suitableSortFn=function(z1,z2) {
229 					var z1w=Math.min(z1.c+z1.w,viewport.w)-z1.c;
230 					var z1h=Math.min(z1.r+z1.h,viewport.h)-z1.r;
231 					var z2w=Math.min(z2.c+z2.w,viewport.w)-z2.c;
232 					var z2h=Math.min(z2.r+z2.h,viewport.h)-z2.r;
233 					return z2w*z2h-z1w*z1h;				
234 				};
235 				break;
236 				default:
237 					$jBlocks("log","Unknown block.pref",block.pref);
238 			}

246 				for(var i=layouts.length-1;i>=0;i--) {

249 						if($.jBlocks.worker.supported)
250 							self.postMessage({
251 								cmd: "layout",
252 								args: {

254 										block: layout.block,
255 									},
256 									misfit: true,
257 								},

259 							});

261 						blocks.unshift(layout.block);

263 						relayoutCount++;
264 					}
265 				}
266 				if(relayoutCount==0) {
267 					layoutData.misfits[block.id]=true;
268 					if($.jBlocks.worker.supported)
269 						self.postMessage({
270 							cmd: "layout",
271 							args: {
272 								layout: {
273 									block: block,
274 								},
275 								misfit: true,
276 							},

278 						});

280 					blocks.unshift(block);
281 			} else {
282 				var alignHoriz="left";
283 				var alignVert="top";
284 				if(block.pref) {
285 					if(block.pref.indexOf("right")>=0)
286 						alignHoriz="right";
287 					if(block.pref.indexOf("bottom")>=0)
288 						alignVert="bottom";
289 					if(block.pref.indexOf("center")>=0) {
290 						alignVert="center";
291 						alignHoriz="center";
292 					}
293 				}

300 					w: block.w,

308 				switch(alignHoriz) {

311 				case "center": layout.c=zone.c+Math.floor((zone.w-block.w)/2); break;
312 				}

322 				layoutData.rmk.cols[layout.c+layout.block.w]=layout.c+layout.block.w;
323 				delete layoutData.misfits[layout.block.id];
324 				if($.jBlocks.worker.supported)
325 					self.postMessage({
326 						cmd: "layout",
327 						args: {
328 							layout: layout,
329 							misfit: false,
330 						},

332 					});

334 				layouts.push(layout);

337 					frameSize.w=layout.c+layout.block.w;
338 				if(frameSize.h<0 || layout.r+layout.block.h>frameSize.h)
339 					frameSize.h=layout.r+layout.block.h;
340

342 			}

373 			layoutData.rmk.rows[layoutData.maxSize.h-viewport.h]=layoutData.maxSize.h-viewport.h;
374 		if(layoutData.maxSize.w>viewport.w)
375 			layoutData.rmk.cols[layoutData.maxSize.w-viewport.w]=layoutData.maxSize.w-viewport.w;
376 	
377 		if(viewport.fillHoles) {
378 			var zones=BuildZones(null);
379 			zones.sort(function(z1,z2) {
380 				return z2.w*z2.h-z1.w*z1.h;
381 			});
382 			//$.jBlocks.worker.log("zones",zones);
383 			
384 			var modified=true;
385 			while(modified) {
386 				modified=false;
387 				for(var zi=0;zi<zones.length && !modified;zi++) {
388 					var zone=zones[zi];
389 					var empty=true;
390 					for(var li=0;li<layouts.length && empty;li++) {
391 						var layout=layouts[li];
392 						if(!(layout.r+layout.h<=zone.r || 
393 								layout.r>=zone.r+zone.h ||
394 								layout.c+layout.w<=zone.c || 
395 								layout.c>=zone.c+zone.w)) {
396 							//$jBlocks("log","Not suitable");
397 							empty=false;
398 							break;
399 						}
400 					}
401 //					if(zone.r+zone.h==viewport.h)
402 //						$.jBlocks.worker.log("empty zone",zone);
403 
404 					if(empty && (zone.c+zone.w<=layoutData.maxSize.w || zone.c+zone.w<=viewport.w) && (zone.r+zone.h<=layoutData.maxSize.h || zone.r+zone.h<=viewport.h)) {
405 						//$.jBlocks.worker.log("empty zone",zone,"blocks",layouts.length);
406 						var candidates=[];
407 						for(var li=0;li<layouts.length;li++) {
408 							var layout=layouts[li];
409 							if(layoutData.misfits[layout.block.id])
410 								continue;
411 							if(layout.block.expandWidth>0 && layout.r>=zone.r && layout.r+layout.h<=zone.r+zone.h) {
412 								if(layout.c+layout.w==zone.c) {
413 									candidates.push({
414 										li: li,
415 										change: {
416 											w: layout.w+zone.w,
417 										},
418 										weight: "expandWidth",
419 									});
420 								}
421 								if(zone.c+zone.w==layout.c) {
422 									candidates.push({
423 										li: li,
424 										change: {
425 											c: layout.c-zone.w,
426 											w: layout.w+zone.w,
427 										},
428 										weight: "expandWidth",
429 									});
430 								}
431 							}
432 							if(layout.block.expandHeight>0 && layout.c>=zone.c && layout.c+layout.w<=zone.c+zone.w) {
433 								if(layout.r+layout.h==zone.r) {
434 									candidates.push({
435 										li: li,
436 										change: {
437 											h: layout.h+zone.h,
438 										},
439 										weight: "expandHeight",									
440 									});
441 								}
442 								if(zone.r+zone.h==layout.r) {
443 									candidates.push({
444 										li: li,
445 										change: {
446 											r: layout.r-zone.h,
447 											h: layout.h+zone.h,
448 										},
449 										weight: "expandHeight",									
450 									});
451 								}
452 							}
453 						}
454 						if(candidates.length>0) {
455 							//$.jBlocks.worker.log("candidates",candidates);
456 							modified=true;
457 							candidates.sort(function(l1,l2) {
458 								var v1=layouts[l1.li].block[l1.weight];
459 								var v2=layouts[l2.li].block[l2.weight];
460 								return v2-v1;
461 							});
462 							var candidate=candidates[0];
463 							var layout=layouts[candidate.li];
464 							$.jBlocks.worker.extend(layout,candidate.change);
465 							if($.jBlocks.worker.supported)
466 								self.postMessage({
467 									cmd: "layout",
468 									args: {
469 										layout: layout,
470 										misfit: false,
471 									},
472 									jBlocksIndex: jBlocksIndex,
473 								});
474 						}
475 					}
476 				}
477 			}
478 		}
479 		
480 		return layouts.concat(invisibleLayouts);
481 	},
