 12 /**
 13  * <p>An Input block displays an HTML input field.</p>
 14  * <p>The constructor should not be invoked directly but {@link JBlocks#newBlock} or {@link JBlocks#addNewBlock} should be used instead.</p>
 15  * @name Input
 16  * @class Define a Input block to be displayed into a JBlocks.
 17  * @augments HtmlElement
 18  */
 19 $.jBlocks("defineBlockClass","Input","HtmlElement",{
 20 	/**
 21 	 * Initializes an Input block.
 22 	 * The method is invoked by the jBlocks core and must never be called directly.
 23 	 * @name Input#init
 24 	 * @function
 25 	 * @param {Object} [layout] see {@link Block#init}.
 26 	 * @param {Object} [options] in addition to base class options {@link Block#init}:
 27 	 * @param {Function} [options.onCR] a function to be called when carriage return is pressed inside the input field.
 28 	 * @param {Function} [options.onKeyDown] a function(keyCode) to be called when a key is pressed.
 29 	 * @param {Function} [options.text=""] the initial text value to display in the input field.
 30 	 */
 31 	init: function() {
 32 		this._super.apply(this,arguments);
 33 		this.options=$.extend({
 34 			onCR: function() {},		
 35 			onKeyDown: function() {},		
 36 			text: "",
 37 		},this.options);
 38 		this.options.tag="input";
 39 		this.options.blockClass="jblocks-block-input";
 40 	},
 41 	onCreate: function() {
 42 		this._super.apply(this);
 43 		var $this=this;
 44 		this.element.val(this.options.text).bind("keydown",function(event) {
 45 			if(event.keyCode==13)
 46 				$this.options.onCR.call($this);
 47 			else
 48 				$this.options.onKeyDown.call($this,event.keyCode);
 49 		});
 50 	},
 51 	display: function() {
 52 		this._super.apply(this,arguments);
 53 		this.element.css({
 54 			"font-size": this.element.height()*0.6+"pt",
 55 		});
 56 	},
 57 	/*
 58 	display: function(geometry) {
 59 		this.input.css({
 60 			margin: 0,
 61 			padding: geometry.height*0.05,
 62 			"font-size": geometry.height*0.4+"pt",
 63 		});
 64 		var extraHeight=this.input.outerHeight(true)-this.input.height();
 65 		var extraWidth=this.input.outerWidth(true)-this.input.width();
 66 		this.input.height(geometry.height-extraHeight);
 67 		this.input.width(geometry.width-extraWidth);
 68 	},
 69 	*/
 70 	/**
 71 	 * Get or set the input field value.
 72 	 * @param {String} [text] if omitted, the function returns the current value of the field, otherwise, sets the input value.
 73 	 * @returns the input text value.
 74 	 */
 75 	val: function() {
 76 		return this.element.val.apply(this.element,arguments);
 77 	},
 78 });
